// Point.java

package genericP;

/********************************************************/
/*         Point フィールドの座標                       */
/********************************************************/

public class Point{

  double x;
  double y;

/***************************************************************************/

  public Point(){
    x = 0.0;
    y = 0.0;
  }

/***************************************************************************/

  public Point(double tx, double ty){
    x = tx;
    y = ty;
  }

/***************************************************************************/

  public void setPoint(double tx, double ty){
    x = tx;
    y = ty;
  }

/***************************************************************************/

  public void setPointX(double tx){
    x = tx;
  }

/***************************************************************************/

  public void setPointY(double ty){
    y = ty;
  }

/***************************************************************************/

  public double getPointX(){
    return x;
  }

/***************************************************************************/

  public double getPointY(){
    return y;
  }

}
